void divide(int s,int d,int &m)
{
int i=s,j=d,pi=0,pj=1;
while(i<j)
  {if(x[i]>x[j])
     {schimb(x[i],x[j]);
      schimb(pi,pj);}
     i=i+pi;
     j=j-pj;}
  m=i;
}

void quicksort(int s,int d)
{int m;
if(s<d)
{
