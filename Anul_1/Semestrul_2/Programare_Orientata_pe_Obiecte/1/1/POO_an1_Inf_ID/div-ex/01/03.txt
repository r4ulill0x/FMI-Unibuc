int sign(int s[50])
{int m=0,v[50];//m este nr de inversiuni ale permutarii s
for(int i=1;i<=n;i++)
  v[i]=0;
for(int k=n;k>0;k--)
  for(int i=1;i<=n;i++)
	 if(s[i]==k)
	   {m+=n-i-v[i];
	    for(int j=1;j<i;j++)
		   v[j]++;
	   break;
	   }
return pow(-1,m);
}
