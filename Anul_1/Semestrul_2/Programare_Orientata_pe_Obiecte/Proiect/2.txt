#include <iostream>
#include <vector>
#include <algorithm>

template <typename T>
class Polinom {
private:
    std::vector<T> coeficienti;

public:
    Polinom() {}

    Polinom(const std::vector<T>& coef) : coeficienti(coef) {
        normalize();
    }

    void normalize() {
        while (!coeficienti.empty() && coeficienti.back() == T(0)) {
            coeficienti.pop_back();
        }
    }

    Polinom operator+(const Polinom& other) const {
        std::vector<T> result(std::max(coeficienti.size(), other.coeficienti.size()), T(0));
        for (size_t i = 0; i < coeficienti.size(); i++) result[i] += coeficienti[i];
        for (size_t i = 0; i < other.coeficienti.size(); i++) result[i] += other.coeficienti[i];
        return Polinom(result);
    }

    Polinom operator-(const Polinom& other) const {
        std::vector<T> result(std::max(coeficienti.size(), other.coeficienti.size()), T(0));
        for (size_t i = 0; i < coeficienti.size(); i++) result[i] += coeficienti[i];
        for (size_t i = 0; i < other.coeficienti.size(); i++) result[i] -= other.coeficienti[i];
        return Polinom(result);
    }

    Polinom operator*(const Polinom& other) const {
        std::vector<T> result(coeficienti.size() + other.coeficienti.size() - 1, T(0));
        for (size_t i = 0; i < coeficienti.size(); i++)
            for (size_t j = 0; j < other.coeficienti.size(); j++)
                result[i + j] += coeficienti[i] * other.coeficienti[j];
        return Polinom(result);
    }

    bool operator==(const Polinom& other) const {
        return coeficienti == other.coeficienti;
    }

    void afisare() const {
        for (size_t i = 0; i < coeficienti.size(); i++) {
            std::cout << coeficienti[i] << "x^" << i;
            if (i + 1 < coeficienti.size()) std::cout << " + ";
        }
        std::cout << std::endl;
    }
};

int main() {
    Polinom<int> p1({1, 2, 3});
    Polinom<int> p2({3, 4, 5});
    
    Polinom<int> suma = p1 + p2;
    Polinom<int> diferenta = p1 - p2;
    Polinom<int> produs = p1 * p2;

    suma.afisare();
    diferenta.afisare();
    produs.afisare();
}
